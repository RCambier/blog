Inferring the proportion of a binomial distribution by observing realization.

fake_observed = sp.stats.bernoulli(0.85).rvs(size=28)

fig, axes = plt.subplots(3, 1, sharex=True, figsize=(20,7))
plt.xlabel('')


# The grid
possible_values_for_p = np.linspace(0,1,100)

# The prior
prior = sp.stats.beta.pdf(possible_values_for_p, a=2, b=2) # The prior distribution. Represents our initial belief about the proportion of the binomial

axes[0].plot(prior)
axes[0].set_title("Prior")


# The likelihood
likelihood = sp.stats.binom.pmf(k=fake_observed.sum(), n=28, p=possible_values_for_p) 

axes[1].plot(likelihood, label="likelihood")
axes[1].set_title("Likelihood")


# The posterior
posterior_unnormalized = prior * likelihood
posterior = posterior_unnormalized / posterior_unnormalized.sum()

axes[2].plot(posterior, label="posterior")
axes[2].set_title("Posterior")
